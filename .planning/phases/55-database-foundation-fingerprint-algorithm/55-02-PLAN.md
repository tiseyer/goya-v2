**File 1: `lib/device/fingerprint.ts`** (per FP-01)

Client-safe utility. Uses Web Crypto API (browser built-in). NO server-only imports.

```typescript
/**
 * Generates a stable device fingerprint using SHA-256.
 * Inputs: screen width, height, color depth, timezone, language.
 * User-Agent is EXCLUDED — it changes on every Chrome release (~6 weeks)
 * and would cause mass re-verification storms.
 */
export async function getDeviceFingerprint(): Promise<string> {
  const raw = [
    screen.width.toString(),
    screen.height.toString(),
    screen.colorDepth.toString(),
    Intl.DateTimeFormat().resolvedOptions().timeZone,
    navigator.language,
  ].join('|')

  const encoded = new TextEncoder().encode(raw)
  const hashBuffer = await crypto.subtle.digest('SHA-256', encoded)
  const hashArray = Array.from(new Uint8Array(hashBuffer))
  return hashArray.map(b => b.toString(16).padStart(2, '0')).join('')
}
```

Key constraints per research:
- Do NOT include `navigator.userAgent` in the hash input (Pitfall 4)
- DO include screen.colorDepth and navigator.language (FP-01 requirement)
- Use `crypto.subtle.digest` (browser built-in, zero bundle cost)
- Return lowercase hex string (64 chars for SHA-256)

**File 2: `lib/device/parseDeviceName.ts`** (per FP-02)

Client-safe utility. Parses userAgent into "Browser on OS" format.

```typescript
/**
 * Parses navigator.userAgent into a human-readable device name.
 * Format: "Browser on OS" — e.g., "Chrome on macOS", "Safari on iOS".
 * Used for admin display in the trusted devices list.
 */
export function parseDeviceName(ua?: string): string {
  const userAgent = ua || (typeof navigator !== 'undefined' ? navigator.userAgent : '')
  if (!userAgent) return 'Unknown Device'

  // Detect browser
  let browser = 'Unknown Browser'
  if (userAgent.includes('Firefox/')) browser = 'Firefox'
  else if (userAgent.includes('Edg/')) browser = 'Edge'
  else if (userAgent.includes('OPR/') || userAgent.includes('Opera')) browser = 'Opera'
  else if (userAgent.includes('Chrome/') && !userAgent.includes('Edg/')) browser = 'Chrome'
  else if (userAgent.includes('Safari/') && !userAgent.includes('Chrome/')) browser = 'Safari'

  // Detect OS
  let os = 'Unknown OS'
  if (userAgent.includes('iPhone') || userAgent.includes('iPad')) os = 'iOS'
  else if (userAgent.includes('Android')) os = 'Android'
  else if (userAgent.includes('Mac OS X') || userAgent.includes('Macintosh')) os = 'macOS'
  else if (userAgent.includes('Windows')) os = 'Windows'
  else if (userAgent.includes('Linux')) os = 'Linux'
  else if (userAgent.includes('CrOS')) os = 'ChromeOS'

  return `${browser} on ${os}`
}
```

**File 3: `lib/device/checkTrustedDevice.ts`** (for Phase 56-57 consumption)

Server-only utility. Uses service-role client. Enforces 90-day rolling window.

```typescript
import { getSupabaseService } from '@/lib/supabase/service'

/**
 * Checks if a device fingerprint is trusted for a given profile.
 * Uses service-role client (bypasses RLS).
 * Returns true only if a matching row exists AND last_used_at is within 90 days.
 *
 * IMPORTANT: This is the defense-in-depth check. The middleware cookie lock
 * is UX-only — every API route that touches user data must call this function
 * independently (CVE-2025-29927 pattern).
 */
export async function checkTrustedDevice(
  profileId: string,
  fingerprint: string
): Promise<boolean> {
  const supabase = getSupabaseService()
  const ninetyDaysAgo = new Date(Date.now() - 90 * 24 * 60 * 60 * 1000).toISOString()

  const { data } = await (supabase as any)
    .from('trusted_devices')
    .select('id')
    .eq('profile_id', profileId)
    .eq('device_fingerprint', fingerprint)
    .gte('last_used_at', ninetyDaysAgo)
    .maybeSingle()

  return Boolean(data)
}
```

Note: The `as any` cast is needed because `trusted_devices` may not be in the generated types yet if Plan 01 hasn't completed. This is safe — the query is constructed manually and the column names are verified against the migration.

**File 1: `app/components/DeviceFingerprintSetter.tsx`** (per FP-03, FP-04)

```typescript
'use client'

import { useEffect } from 'react'
import { getDeviceFingerprint } from '@/lib/device/fingerprint'
import { parseDeviceName } from '@/lib/device/parseDeviceName'

const COOKIE_NAME = 'goya_device_fp'
const DEVICE_NAME_COOKIE = 'goya_device_name'
const MAX_AGE = 365 * 24 * 60 * 60 // 365 days in seconds
const SECURE = process.env.NODE_ENV === 'production' ? '; Secure' : ''

/**
 * Sets the device fingerprint and device name cookies on every page load.
 * - goya_device_fp: SHA-256 hash of screen dimensions + timezone + language (NO userAgent)
 * - goya_device_name: human-readable "Browser on OS" parsed from userAgent
 *
 * Both cookies: SameSite=Lax, httpOnly=false, 365-day maxAge.
 * httpOnly=false is required — the fingerprint must be readable by client JS
 * and sent with the /auth/callback redirect (SameSite=Lax allows this).
 *
 * Only writes if the cookie is absent to avoid unnecessary cookie writes.
 */
export function DeviceFingerprintSetter() {
  useEffect(() => {
    const existingFp = document.cookie
      .split('; ')
      .find(row => row.startsWith(`${COOKIE_NAME}=`))

    if (!existingFp) {
      getDeviceFingerprint().then(fp => {
        document.cookie = `${COOKIE_NAME}=${fp}; path=/; max-age=${MAX_AGE}; SameSite=Lax${SECURE}`
      })
    }

    // Always update device name (it can change with browser updates, which is fine for display)
    const deviceName = parseDeviceName()
    document.cookie = `${DEVICE_NAME_COOKIE}=${encodeURIComponent(deviceName)}; path=/; max-age=${MAX_AGE}; SameSite=Lax${SECURE}`
  }, [])

  return null // No visible UI
}
```

Key constraints:
- Cookie name is `goya_device_fp` (NOT `sb-` prefix — would conflict with Supabase SSR, per Pitfall 6)
- `httpOnly: false` is required (per FP-03) — client JS must write the cookie
- `SameSite=Lax` (NOT Strict — fingerprint must be sent on /auth/callback redirect, per Pitfall 6)
- `max-age=31536000` (365 days per FP-03)
- Only writes fingerprint if cookie is absent (prevents unnecessary rewrites)
- `goya_device_name` cookie stores the human-readable name for use by verify-code API route
- Returns null — no visible UI
