export type HttpMethod = 'GET' | 'POST' | 'PATCH' | 'DELETE'
export type AuthType = 'none' | 'read' | 'write' | 'admin'
export type EndpointCategory = 'Health' | 'Users' | 'Events' | 'Courses' | 'Credits' | 'Verifications' | 'Analytics' | 'Add-ons' | 'Admin' | 'Webhooks'

export interface Endpoint {
  method: HttpMethod
  path: string
  auth: AuthType
  description: string
  category: EndpointCategory
}

export const ENDPOINT_CATEGORIES: EndpointCategory[] = [
  'Health', 'Users', 'Events', 'Courses', 'Credits',
  'Verifications', 'Analytics', 'Add-ons', 'Admin', 'Webhooks',
]

**2. Create `app/admin/api-keys/EndpointsTab.tsx`**

A `'use client'` component matching the SecretsTab pattern:

- Props: none (data is imported from registry, not passed as props — it is static)
- State: `searchQuery` (string), `selectedCategory` (EndpointCategory | 'All')
- Filter logic: client-side `.filter()` combining category match + search match (search against path and description, case-insensitive)
- Group filtered endpoints by category using `ENDPOINT_CATEGORIES` order
- Only render category groups that have matching endpoints after filtering

**UI structure (matching SecretsTab design tokens):**

Toolbar row (same flex layout as SecretsTab toolbar):
- Category filter pills: "All" + each category from ENDPOINT_CATEGORIES. Use `rounded-full text-xs font-semibold px-3 py-1.5` classes. Active pill: `bg-[#00B5A3]/10 text-[#00B5A3]`. Inactive: `bg-slate-100 text-slate-600 hover:bg-slate-200`.
- Search input: same pattern as SecretsTab — `text-sm` input with magnifying glass SVG icon, placeholder "Search endpoints...", `border border-[#E5E7EB] rounded-lg px-3 py-2 pl-9`
- Endpoint count badge: show "{N} endpoints" in `text-xs text-slate-400` on the right

For each category group with results:
- Category header: `text-sm font-semibold text-[#1B3A5C] mb-2 mt-6 first:mt-0` with category name + count in parentheses
- Endpoint rows inside a `bg-white rounded-xl border border-[#E5E7EB] divide-y divide-[#E5E7EB]` container:
  - Each row: `px-4 py-3 flex items-start gap-3`
  - Method badge: fixed-width `w-16 inline-flex justify-center rounded text-xs font-bold py-1` with colors:
    - GET: `bg-emerald-100 text-emerald-700`
    - POST: `bg-blue-100 text-blue-700`
    - PATCH: `bg-amber-100 text-amber-700`
    - DELETE: `bg-red-100 text-red-700`
  - Path: `font-mono text-sm text-[#1B3A5C]` — highlight `:param` segments with `text-slate-400`
  - Auth badge: `text-xs rounded-full px-2 py-0.5` — none: `bg-slate-100 text-slate-500`, read: `bg-emerald-50 text-emerald-600`, write: `bg-blue-50 text-blue-600`, admin: `bg-red-50 text-red-600`
  - Description: `text-sm text-[#6B7280]`

Empty state when no results match filters: "No endpoints match your search." centered text in a bordered card (same style as EndpointsPlaceholder).

Path rendering helper: split path by '/' and wrap segments starting with ':' in a span with `text-slate-400` class to visually distinguish parameters.
