```typescript
export type FlowElement =
  | FlowElementInfoText      // type: 'info_text', content: string
  | FlowElementShortText     // type: 'short_text'
  | FlowElementLongText      // type: 'long_text'
  | FlowElementSingleChoice  // type: 'single_choice', options: { label, value }[]
  | FlowElementMultiChoice   // type: 'multi_choice', options: { label, value }[]
  | FlowElementDropdown      // type: 'dropdown', options: { label, value }[]
  | FlowElementImageUpload   // type: 'image_upload'
  | FlowElementImage         // type: 'image', src: string, alt: string
  | FlowElementVideo         // type: 'video', url: string

// All extend FlowElementBase: { element_key, label, required, help_text }

export interface FlowStep {
  id: string;
  flow_id: string;
  position: number;
  title: string | null;
  elements: FlowElement[];
  schema_version: number;
  created_at: string;
}

export interface FlowBranch {
  id: string;
  step_id: string;
  element_key: string;
  answer_value: string;
  target_step_id: string;
}

export type FlowWithSteps = Flow & {
  steps: (FlowStep & { branches: FlowBranch[] })[];
};

export type CreateStepInput = { flow_id: string; position?: number; title?: string | null; elements?: FlowElement[] };
export type UpdateStepInput = Partial<Omit<CreateStepInput, 'flow_id'>>;
```

API routes:
```
GET    /api/admin/flows/[id]                       → FlowWithSteps
POST   /api/admin/flows/[id]/steps                 → FlowStep (create step)
PATCH  /api/admin/flows/[id]/steps/[stepId]        → FlowStep (update step)
DELETE /api/admin/flows/[id]/steps/[stepId]         → { success: true }
PUT    /api/admin/flows/[id]/steps/reorder          → FlowStep[] (reorder by stepIds array)
```

   ```typescript
   import { create } from 'zustand';
   import type { Flow, FlowStep, FlowBranch, FlowElement, FlowWithSteps } from './types';

   interface EditorState {
     // Data
     flow: Flow | null;
     steps: (FlowStep & { branches: FlowBranch[] })[];
     // UI state
     selectedStepId: string | null;
     isDirty: boolean;
     isSaving: boolean;
     // Actions
     initializeFlow: (flowWithSteps: FlowWithSteps) => void;
     selectStep: (stepId: string | null) => void;
     addStep: (step: FlowStep & { branches: FlowBranch[] }) => void;
     removeStep: (stepId: string) => void;
     reorderSteps: (stepIds: string[]) => void;
     updateStepElements: (stepId: string, elements: FlowElement[]) => void;
     updateStepTitle: (stepId: string, title: string) => void;
     updateFlow: (partial: Partial<Flow>) => void;
     setDirty: (dirty: boolean) => void;
     setSaving: (saving: boolean) => void;
   }
   ```
   Implement all actions. `initializeFlow` sets flow, steps, selectedStepId to first step. `selectStep` sets selectedStepId. `addStep` appends and selects. `removeStep` removes and selects next/previous. `reorderSteps` reorders by provided ID array. `updateStepElements` replaces elements array for a step and sets isDirty=true.
