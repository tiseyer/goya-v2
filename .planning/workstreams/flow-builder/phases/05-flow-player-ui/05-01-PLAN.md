export type FlowDisplayType = 'modal' | 'fullscreen' | 'top_banner' | 'bottom_banner' | 'notification';

export interface FlowElementBase {
  element_key: string;
  label: string;
  required: boolean;
  help_text: string | null;
}

export type FlowElement =
  | FlowElementInfoText    // type: 'info_text', content: string
  | FlowElementShortText   // type: 'short_text'
  | FlowElementLongText    // type: 'long_text'
  | FlowElementSingleChoice // type: 'single_choice', options: {label, value}[]
  | FlowElementMultiChoice  // type: 'multi_choice', options: {label, value}[]
  | FlowElementDropdown     // type: 'dropdown', options: {label, value}[]
  | FlowElementImageUpload  // type: 'image_upload'
  | FlowElementImage        // type: 'image', src: string, alt: string
  | FlowElementVideo        // type: 'video', url: string

export interface ActiveFlowResponse {
  flow: Omit<Flow, 'conditions'>;
  steps: (FlowStep & { branches: FlowBranch[] })[];
  response: FlowResponse | null;
}

// Flow has: display_type, modal_dismissible, modal_backdrop
// FlowResponse has: last_step_id, responses: Record<string, unknown>

    Props: `{ currentStep: number; totalSteps: number }`.
    Render a `div` with `h-1.5 rounded-full bg-gray-200 dark:bg-gray-700 overflow-hidden` container. Inner div with `h-full bg-[var(--color-primary)] rounded-full` and `style={{ width: \`${(currentStep / totalSteps) * 100}%\` }}`. Add `transition-all duration-300 ease-out` for smooth progression.

    **FlowNavigation.tsx**: Back/Next/Complete button row.
    Props: `{ onBack: () => void; onNext: () => void; onComplete: () => void; canGoBack: boolean; canGoNext: boolean; isLastStep: boolean; isSubmitting: boolean }`.
    - Back button: `text-gray-500 hover:text-gray-700` with ChevronLeft icon, hidden when `!canGoBack`
    - Next button: `bg-[var(--color-primary)] text-white rounded-lg px-6 py-2.5 font-medium hover:opacity-90 disabled:opacity-50 disabled:cursor-not-allowed transition`
    - On last step, Next becomes "Complete" with a CheckCircle icon
    - Show spinner (lucide-react Loader2 with animate-spin) when isSubmitting
    - Next/Complete is disabled when `!canGoNext || isSubmitting`

    **FlowPlayer.tsx**: Core orchestrator. This is the main component.
    State (useState):
    - `activeFlow: ActiveFlowResponse | null` — fetched from API
    - `currentStepIndex: number` — which step is being shown
    - `answers: Record<string, unknown>` — element_key -> value for ALL steps (merged from response.responses on load)
    - `isSubmitting: boolean`
    - `isLoading: boolean`
    - `error: string | null`

    On mount (useEffect):
    1. `fetch('/api/flows/active?trigger=login')` with credentials: 'include'
    2. Parse ActiveFlowResponse. If `flow` is null, render nothing.
    3. If `response?.last_step_id` exists, find its index in steps array and set `currentStepIndex` to the NEXT step (resume from last completed). Load `response.responses` into `answers`.

    Step navigation:
    - `handleNext()`: Validate required fields for current step elements. If valid, call `POST /api/flows/[id]/respond` with `{ step_id: currentStep.id, answers: currentStepAnswers }`. The `answers` sent should be only the element_key->value pairs for the current step. Include step actions from `steps[currentStepIndex]` if present. On success, advance `currentStepIndex`. If action results include `redirect`, do `window.location.href = url`. If `success_popup`, show a brief toast/alert.
    - `handleBack()`: Decrement `currentStepIndex` (no API call — answers are already saved).
    - `handleComplete()`: Call `POST /api/flows/[id]/complete`. Set `activeFlow` to null (unmounts player).
    - `handleDismiss()`: Set `activeFlow` to null. (Only callable from dismissible modals.)

    Required field validation:
    - For current step, iterate elements. If `element.required === true`, check `answers[element.element_key]` is truthy (non-empty string, non-empty array, non-null).
    - `canGoNext` = all required fields have values AND not isSubmitting.

    Branch resolution:
    - After submitting a step, check if any branch in `currentStep.branches` matches the answers. If a branch matches (branch.element_key matches element_key and branch.answer_value matches the answer value), set `currentStepIndex` to the index of the branch's `target_step_id` in the steps array. Otherwise advance sequentially.

    Render: Based on `activeFlow.flow.display_type`, wrap step content in the appropriate display component (FlowPlayerModal or FlowPlayerFullscreen for now — banner/notification in Plan 02). Step content = FlowProgress + step title + ElementRenderer for each element + FlowNavigation.

    **FlowPlayerModal.tsx (PLAYER-01, PLAYER-02, PLAYER-03)**: Modal overlay wrapper.
    Props: `{ children: ReactNode; dismissible: boolean; backdrop: 'blur' | 'dark' | 'none' | null; onDismiss: () => void }`.
    - Use `createPortal` to render into `document.body` (z-index safety per research pitfalls)
    - Backdrop: `fixed inset-0 z-[9999]`. If backdrop='blur': `backdrop-blur-sm bg-black/30`. If 'dark': `bg-black/50`. If 'none' or null: `bg-black/20`.
    - Modal container: `fixed inset-0 z-[10000] flex items-center justify-center p-4`. Inner: `bg-white dark:bg-gray-900 rounded-2xl shadow-2xl w-full max-w-lg max-h-[85vh] overflow-y-auto p-6 relative`.
    - If `dismissible`: render X button (lucide-react X) top-right `absolute top-4 right-4`. Backdrop click calls `onDismiss`.
    - If NOT dismissible: no X button. Backdrop click triggers shake animation — use motion `animate={{ x: [0, -8, 8, -4, 4, 0] }}` with `transition={{ duration: 0.4 }}` on the modal container div. Use a `shakeKey` state that increments to re-trigger.
    - Wrap modal container in `<motion.div>` with `initial={{ opacity: 0, scale: 0.95 }} animate={{ opacity: 1, scale: 1 }} exit={{ opacity: 0, scale: 0.95 }}`. Wrap outer in `<AnimatePresence>`.
